📁 agent.py

Main agent graph. Binds Gemini to 6 tools, defines the ReAct node, builds and compiles the LangGraph StateGraph.

📁 server.py

FastMCP tool server. Exposes 6 database tools via the MCP protocol. Each tool performs a specific PostgreSQL operation.

📁 state.py

Defines the typed AgentState for LangGraph. Holds raw logs, evidence chain, recovery plan, risk score, and message history.

# Initialize Gemini LLM
llm = ChatGoogleGenerativeAI(
    model="gemini-2.5-flash",
    temperature=0.4
)

# Bind 6 MCP tools to the LLM
llm_with_tools = llm.bind_tools(tools)

# Agent node — injects system prompt once
def agent_node(state: AgentState):
    messages = state["messages"]
    if not messages or not isinstance(messages[0], SystemMessage):
        messages = [SystemMessage(content=SYSTEM_PROMPT)] + list(messages)
    response = llm_with_tools.invoke(messages)
    return {"messages": [response]}

# Build the StateGraph

state.py — Agent Memory TypedDict 
class Evidence(BaseModel):
    log_line: str   # Raw log text
    source:   str   # app / db / infra / monitoring
    timestamp:str   # ISO format
    reasoning:str   # Why this log proves root cause

class RecoveryPlan(BaseModel):
    steps:          List[str]
    estimated_risk: int         # 0–100
    rollback_plan:  str

# The LangGraph state — agent's working memory
class AgentState(TypedDict):
    raw_logs:       str
    evidence_chain: Annotated[List[Evidence], operator.add]
    root_cause:     Optional[str]
    recovery_plan:  Optional[RecoveryPlan]
    final_risk_score:Optional[int]
    ticket_status:  str
    messages:       Annotated[List[str], operator.add]
server.py — FastMCP Tool: finalize_incident() FastMCP · PostgreSQL 
@mcp.tool()
def finalize_incident(
    incident_id: str,  recovery_plan: str,
    risk_score:   int,  agent_notes:   str,  app_id: str = ""
) -> str:
    conn = get_db_connection()
    try:
        cur = conn.cursor()
        # Update incident: ANALYZING → OPEN, write plan + score
        cur.execute("""
            UPDATE incidents
            SET status=%s, risk_score=%s, recovery_plan=%s, agent_notes=%s
            WHERE incident_id=%s
        """, ("OPEN", risk_score, recovery_plan, agent_notes, incident_id))
        # Log the finalization action for audit
        cur.execute("""
            INSERT INTO agent_actions (incident_id, app_id, action_type, input_params, output_result, agent_reasoning)
            VALUES (%s, %s, %s, %s, %s, %s)
        """, (incident_id, app_id or None, "finalize_incident",
               json.dumps({"risk_score": risk_score}), json.dumps({"status": "success"}), agent_notes))
        conn.commit()
        return f"✅ Incident {incident_id} finalized with risk score {risk_score}."
    except Exception as e:
        conn.rollback()
        return f"❌ Error: {str(e)}"
